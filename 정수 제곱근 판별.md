## Solution 1
<pre>
<code>
func solution(_ n:Int64) -> Int64 {
    
    var x = 1
    
    while x * x <= n {
        if x * x == n {
                   return Int64((x+1)*(x+1))
        }
        x += 1
    }
    return -1
}
</code>

func solution(_ n:Int64) -> Int64 {
    let t = Int64(sqrt(Double(n)))
    return t * t == n ? (t+1)*(t+1) : -1
