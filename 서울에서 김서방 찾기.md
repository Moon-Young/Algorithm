## Solution 1
<pre>
<code>
func solution(_ seoul:[String]) -> String {
     
    for i in 0..'<'seoul.count {
        if "Kim" == seoul[i] {
            return "김서방은 \(i)에 있다"
        }
    }
    
    return "김서방은 없다."
}
</code>
