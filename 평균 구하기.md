## Solution 1
<pre>
<code>
func solution(_ arr:[Int]) -> Double {
 
    var sumArr = 0
    
    for i in 0..'<'arr.count {
  
        sumArr += arr[i]   
         
    }
    return Double(sumArr) / Double(arr.count)
}
</pre>
</code>

## Solution 2
<pre>
<code>
func solution(_ arr:[Int]) -> Double {
 
 return Double(arr.reduce(0, +)) / Double(arr.count)
         
}
