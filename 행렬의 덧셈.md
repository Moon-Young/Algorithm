var sumResult: [[Int]] = [[]]

func solution(_ arr1:[[Int]], _ arr2:[[Int]]) -> [[Int]] {

    sumResult.removeAll()
    var sumArray:[Int] = []

    for i in 0..<arr1.count {
        for j in 0..<arr1[i].count {
            print("i:\(i), j:\(j)")
            print(arr1[i].count)
            sumArray.append(arr1[i][j] + arr2[i][j])
            print(sumArray)
        }
        sumResult.append(sumArray)
        sumArray.removeAll()
    }

    print(sumResult)
    return sumResult
}
